"Write a modular, clean ES6 vanilla JavaScript module to power an image carousel. The script should:

Target the slide elements using 'slide' class and active transition state.
Implement event listeners for both the 'Prev' and 'Next' buttons.
Keep track of the current slide index.
Create an updateCarousel() function that handles updating the visual DOM state.
Ensure that when reaching the last slide, clicking 'Next' wraps around to index 0, and clicking 'Prev' at index 0 wraps to the last slide.
Do not use any libraries or frameworks. Provide clean, commented JavaScript code."`

---------------------------

document.addEventListener("DOMContentLoaded", () => {
    const slides = document.querySelectorAll(".slide");
    const prevBtn = document.getElementById("prevBtn");
    const nextBtn = document.getElementById("nextBtn");
    let currentIndex = 0;

    function updateCarousel() {
        slides.forEach((slide, index) => {
            if (index === currentIndex) {
                slide.classList.add("active");
            } else {
                slide.classList.remove("active");
            }
        });
    }

    nextBtn.addEventListener("click", () => {
        currentIndex = (currentIndex + 1) % slides.length; // Modular Wrap Logic
        updateCarousel();
    });

    prevBtn.addEventListener("click", () => {
        currentIndex = (currentIndex - 1 + slides.length) % slides.length; // Negative wrap correction
        updateCarousel();
    });
});
